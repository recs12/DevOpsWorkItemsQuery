SELECT
    [System.Id],
    [System.WorkItemType],
    [System.Title],
    [System.AssignedTo],
    [System.State],
    [System.Tags]
FROM workitems
WHERE
    (
        [Source].[System.IterationPath] = '[PTSA - LiveOrder Projects]\Team ALBP'
        AND [Source].[System.WorkItemType] <> ''
    )
ORDER BY System.ID asc
